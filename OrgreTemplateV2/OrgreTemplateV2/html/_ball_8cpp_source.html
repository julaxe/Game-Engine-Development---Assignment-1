102enum CollisionPlanes Ball::checkCollisionWithPaddle()
103{
104 //get center point circle first

107 //calculate AABB info (center, half-extents);

109 Ogre::Vector3 aabb_center = _paddle->GetNode()->getPosition();
110
111 //get difference vectro beween both centers;
112 Ogre::Vector3 difference = center - aabb_center;
113 Ogre::Vector3 clamped = Ogre::Vector3(Ogre::Math::Clamp<Real>(difference.x, -aabb_half_entents.x, aabb_half_entents.x),
114 Ogre::Math::Clamp<Real>(difference.y, -aabb_half_entents.y, aabb_half_entents.y), 0.0f);
115
116 //add value to aabb center and we get the value of box closest to circle
117 Ogre::Vector3 closest = aabb_center + clamped;
118 //retrieve vector between center circle and closest point AABB and check if lenght <= radius
119 difference = closest - center;
120
121 if (difference.length() < radius)
